 21class SpringTorque(chrono.TorqueFunctor):
 22
 23    def __init__(self, spring_coef, damping_coef, rest_angle):
 24        super(SpringTorque, self).__init__()
 25        self.spring_coef = spring_coef
 26        self.damping_coef = damping_coef
 27        self.rest_angle = rest_angle
 28
 29    def evaluate(
 30            self,  #
 31            time,  # current time
 32            angle,  # relative angle of rotation
 33            vel,  # relative angular speed
 34            link):  # back-pointer to associated link
 35        torque = 0
 36        if self.spring_coef > 10**-3:
 37            torque = -self.spring_coef * \
 38                (angle - self.rest_angle) - self.damping_coef * vel
 39        else:
 40            torque = -self.damping_coef * vel
 41        return torque

 44class ContactReporter(chrono.ReportContactCallback):
 45
 46    def __init__(self, chrono_body):
 47        """Create a sensor of contact normal forces for the body
 48
 49        Args:
 50            chrono_body (ChBody): The body on which the sensor is install
 51        """
 52        self._body = chrono_body
 53        self.__current_normal_forces = None
 54        self.__list_normal_forces = []
 55        super().__init__()
 56
 57    def OnReportContact(
 58        self,
 59        pA,  # contact pA 
 60        pB,  # contact pB 
 61        plane_coord,  # contact plane coordsystem (A column 'X' is contact normal) 
 62        distance,  # contact distance 
 63        eff_radius,  # effective radius of curvature at contact 
 64        cforce,  # react.forces (if already computed). In coordsystem 'plane_coord' 
 65        ctorque,  # react.torques, if rolling friction (if already computed). 
 66        modA,  # model A (note: some containers may not support it and could be nullptr) 
 67        modB):  # model B (note: some containers may not support it and could be nullptr)
 68        bodyA = chrono.CastToChBody(modA)
 69        bodyB = chrono.CastToChBody(modB)
 70        if (bodyA == self._body) or (bodyB == self._body):
 71            self.__current_normal_forces = cforce.x
 72            self.__list_normal_forces.append(cforce.x)
 73        return True
 74
 75    def is_empty(self):
 76        return len(self.__list_normal_forces) == 0
 77
 78    def list_clear(self):
 79        self.__list_normal_forces.clear()
 80
 81    def get_normal_forces(self):
 82        return self.__current_normal_forces
 83

 88class ChronoBody(BlockBody, ABC):
 89    """Abstract class, that interpreting nodes of a robot body part in a physics engine (`pychrono`).
 90    
 91    Attributes:
 92        body (pychrono.ChBody): Pychrono object of the solid body. It defines visualisation, collision shape, position on the world frame and etc in simulation system.
 93        builder (pychrono.ChSystem): Pychrono object of system, which hosts the body.
 94
 95    Args:
 96        builder (pychrono.ChSystem): Arg sets the system, which hosth the body
 97        body (pychrono.ChBody): Solid body define nodes of the body part in the physics system
 98        in_pos_marker (pychrono.ChVectorD): Arg defines position input frame the body
 99        out_pos_marker (chrono.ChVectorD): Arg defines position output frame the body
100        random_color (bool): Flag of the random color of the body
101        is_collide (bool, optional): Flag of collision body with other objects in system.. Defaults to True.
102    """
103
104    def __init__(self,
105                 builder: chrono.ChSystem,
106                 body: chrono.ChBody,
107                 in_pos_marker: chrono.ChVectorD,
108                 out_pos_marker: chrono.ChVectorD,
109                 random_color: bool,
110                 is_collide: bool = True):
111        """Abstract class of interpretation of nodes of a robot body part in a physics engine
112        
113        Initlization adds body in system, creates input and output marker of the body and sets them. Also,it initilize object of the contact reporter 
114        """
115        super().__init__(builder)
116        self.body = body
117        self.builder.Add(self.body)
118
119        # Create markers aka RefFrames
120        input_marker = chrono.ChMarker()
121        out_marker = chrono.ChMarker()
122        transformed_out_marker = chrono.ChMarker()
123
124        input_marker.SetMotionType(chrono.ChMarker.M_MOTION_KEYFRAMED)
125        out_marker.SetMotionType(chrono.ChMarker.M_MOTION_KEYFRAMED)
126        transformed_out_marker.SetMotionType(chrono.ChMarker.M_MOTION_KEYFRAMED)
127
128        self.body.AddMarker(input_marker)
129        self.body.AddMarker(out_marker)
130        self.body.AddMarker(transformed_out_marker)
131        self.body.GetCollisionModel().SetDefaultSuggestedEnvelope(0.001)
132        self.body.GetCollisionModel().SetDefaultSuggestedMargin(0.0005)
133        self.body.SetCollide(is_collide)
134
135        input_marker.SetPos(in_pos_marker)
136        out_marker.SetPos(out_pos_marker)
137        # Calc SetPos
138        transformed_out_marker.SetCoord(out_marker.GetCoord())
139
140        self._ref_frame_in = input_marker
141        self._ref_frame_out = out_marker
142        self.transformed_frame_out = transformed_out_marker
143
144        # Normal Forces
145        self.__contact_reporter = ContactReporter(self.body)
146
147        if random_color:
148            rgb = [random.random(), random.random(), random.random()]
149            rgb[int(random.random() * 2)] *= 0.2
150            self.body.GetVisualShape(0).SetColor(chrono.ChColor(*rgb))
151
152    def _build_collision_model(self, struct_material, width, length):
153        """Build collision model of the block on material width and length
154
155        Args:
156            struct_material (Material): Dataclass of material body
157            width (flaot): Width of the box
158            length (float): Length of the box
159        """
160        chrono_object_material = struct_material2object_material(struct_material)
161
162        self.body.GetCollisionModel().ClearModel()
163        self.body.GetCollisionModel().AddBox(chrono_object_material, width / 2, length / 2,
164                                             width / 2)
165        self.body.GetCollisionModel().BuildModel()
166
167    def move_to_out_frame(self, in_block: Block):
168        """Move the input frame body to output frame position input block
169
170        Args:
171            in_block (Block): The block defines relative movming to output frame
172        """
173        self.builder.Update()
174        local_coord_in_frame = self._ref_frame_in.GetCoord()
175        abs_coord_out_frame = in_block.transformed_frame_out.GetAbsCoord()
176
177        trans = chrono.ChFrameD(local_coord_in_frame)
178        trans = trans.GetInverse()
179        trans = trans.GetCoord()
180        coord = abs_coord_out_frame * trans
181
182        self.body.SetCoord(coord)
183
184    def make_fix_joint(self, in_block):
185        """Create weld joint (fixing relative posiotion and orientation) between input block and the body
186
187        Args:
188            in_block (Block): The block which define relative fixing position and orientation the body in system
189        """
190        fix_joint = chrono.ChLinkMateFix()
191        fix_joint.Initialize(in_block.body, self.body)
192        self.builder.Add(fix_joint)
193
194    def reset_transformed_frame_out(self):
195        """Reset all transforms output frame of the body and back to initial state
196        """
197        self.transformed_frame_out.SetCoord(self._ref_frame_out.GetCoord())
198
199    def apply_transform(self, in_block: BlockTransform):
200        """Aplied input tranformation to the output frame of the body
201
202        Args:
203            in_block (BlockTransform): The block which define transormations
204        """
205        self.reset_transformed_frame_out()
206        frame_coord = self.transformed_frame_out.GetCoord()
207        frame_coord = frame_coord * in_block.transform
208        self.transformed_frame_out.SetCoord(frame_coord)
209
210    @property
211    def ref_frame_in(self) -> chrono.ChMarker:
212        """Return the input frame of the body
213
214        Returns:
215            pychrono.ChMarker: The input frame of the body
216        """
217        return self._ref_frame_in
218
219    @property
220    def normal_force(self) -> float:
221        """Return value normal forces of random collision point
222
223        Returns:
224            float: Value normal forces of random collision point
225        """
226        self.builder.GetContactContainer().ReportAllContacts(self.__contact_reporter)
227        return self.__contact_reporter.get_normal_forces()
228
229    @property
230    def list_n_forces(self) -> list:
231        """Return a list of all the contact forces
232
233        Returns:
234            list: List normal forces of all the contacts points
235        """
236        container = self.builder.GetContactContainer()
237        contacts = container.GetNcontacts()
238        if contacts:
239            self.__contact_reporter.list_clear()
240            container.ReportAllContacts(self.__contact_reporter)
241        return self.__contact_reporter.get_list_n_forces()

471class ChronoRevolveJoint(BlockBridge):
472    """The class representing revolute joint object in `pychrono` physical engine. It is the embodiment of joint nodes from the mechanism graph in simulation
473    
474
475        Args:
476            builder (pychrono.ChSystem): Arg sets the system, which hosth the body
477            axis (Axis, optional): Define rotation axis. Defaults to Axis.Z.
478            type_of_input (InputType, optional): Define type of input joint control. Defaults to InputType.POSITION. Instead of, can changes to torque, that more realistic.
479            stiffness (float, optional): Optional arg add a spring with `stiffness` to joint. Defaults to 0.
480            damping (float, optional): Optional arg add a dempher to joint. Defaults to 0.
481            equilibrium_position (float, optional): Define equilibrium position of the spring. Defaults to 0.
482            
483        Attributes:
484            joint (pychrono.ChLink): Joint define nodes of the joint part in the system
485            axis (Axis): The axis of the rotation
486            input_type (InputType): The type of input
487    """
488
489    class InputType(str, Enum):
490        TORQUE = {"Name": "Torque", "TypeMotor": chrono.ChLinkMotorRotationTorque}
491        VELOCITY = {"Name": "Speed", "TypeMotor": chrono.ChLinkMotorRotationSpeed}
492        POSITION = {"Name": "Angle", "TypeMotor": chrono.ChLinkMotorRotationAngle}
493        UNCONTROL = {"Name": "Uncontrol", "TypeMotor": chrono.ChLinkRevolute}
494
495        def __init__(self, vals):
496            self.num = vals["Name"]
497            self.motor = vals["TypeMotor"]
498
499    class Axis(str, Enum):
500        # Z is default rotation axis
501        Z = chrono.ChQuaternionD(1, 0, 0, 0)
502        Y = chrono.Q_ROTATE_Z_TO_Y
503        X = chrono.Q_ROTATE_Z_TO_X
504
505    def __init__(self,
506                 builder: chrono.ChSystem,
507                 axis: Axis = Axis.Z,
508                 type_of_input: InputType = InputType.POSITION,
509                 stiffness: float = 0.,
510                 damping: float = 0.,
511                 equilibrium_position: float = 0.):
512        super().__init__(builder=builder)
513        self.joint = None
514        self.axis = axis
515        self.input_type = type_of_input
516        self._ref_frame_out = chrono.ChCoordsysD()
517        # Spring Damper params
518        self._joint_spring = None
519        self._torque_functor = None
520        self.stiffness = stiffness
521        self.damping = damping
522        self.equilibrium_position = equilibrium_position
523
524    def connect(self, in_block: ChronoBody, out_block: ChronoBody):
525        """Joint is connected two bodies
526        
527        If we have two not initialize joints engine crash
528
529        Args:
530            in_block (ChronoBody): Slave body to connect
531            out_block (ChronoBody): Master body to connect
532        """        
533        self.joint = self.input_type.motor()
534        self.joint.Initialize(in_block.body, out_block.body, True, in_block.transformed_frame_out,
535                              out_block.ref_frame_in)
536        self.builder.AddLink(self.joint)
537
538        if (self.stiffness != 0) or (self.damping != 0):
539            self._add_spring_damper(in_block, out_block)
540
541    def apply_transform(self, in_block):
542        """Aplied input tranformation to the output frame of the body
543
544        Args:
545            in_block (BlockTransform): The block which define transormations
546        """
547        self.transformed_frame_out = self._ref_frame_out * in_block.transform
548
549    def _add_spring_damper(self, in_block: ChronoBody, out_block: ChronoBody):
550        self._joint_spring = chrono.ChLinkRSDA()
551        self._joint_spring.Initialize(in_block.body, out_block.body, False,
552                                     in_block.transformed_frame_out.GetAbsCoord(),
553                                     out_block.ref_frame_in.GetAbsCoord())
554        self._torque_functor = SpringTorque(self.stiffness, self.damping, self.equilibrium_position)
555        self._joint_spring.RegisterTorqueFunctor(self._torque_functor)
556        self.builder.Add(self._joint_spring)
