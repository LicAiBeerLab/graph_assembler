 49class GraphEnvironment():
 50    def __init__(self, initilize_graph, rules, max_numbers_rules_non_terminal = 20):
 51        """Class of "environment" of graph grammar 
 52
 53        Args:
 54            initilize_graph (GraphGrammar): Initial state of the graph
 55            rules (list[Rule]): List of rules 
 56            max_numbers_rules_non_terminal (int): Max amount of non-terminal rules. Defaults to 20.
 57        """        
 58        self.init_graph = deepcopy(initilize_graph)
 59        self.graph = deepcopy(initilize_graph)
 60        self._actions = [RuleAction(r) for r in rules] if not rules is None else None
 61        self.max_actions_not_terminal = max_numbers_rules_non_terminal
 62        self.current_player = 1
 63        self.reward = 0
 64        self.counter_action = 0
 65        
 66    # Need override for mcts libary
 67    
 68    def getCurrentPlayer(self):
 69        return self.current_player
 70    
 71
 72    
 73    def getPossibleActions(self):
 74        """Getter possible actions for current state
 75        """
 76        def filter_exist_node(action):
 77            out = False
 78            flags_max_actions = self.counter_action >= self.max_actions_not_terminal
 79            if action.get_replaced_node().label == node:
 80                if action.is_terminal():
 81                    out = True
 82                if not flags_max_actions:
 83                    out = True
 84                
 85            return out
 86        label_nodes = {node[1]["Node"].label for node in self.graph.nodes.items()}
 87        possible_actions = []
 88        for node in label_nodes:
 89            possible_actions += set(filter(filter_exist_node, self._actions))
 90        
 91        return possible_actions
 92        
 93    def takeAction(self, action):
 94        """Take action and return new state environment
 95
 96        Args:
 97            action (RuleAction): Action to take
 98
 99        Returns:
100            GraphEnvironment: New state environment after action taken
101        """
102        rule_action = action.get_rule
103        new_state = deepcopy(self)
104        new_state.graph.apply_rule(rule_action)
105        if not action.is_terminal():
106            new_state.counter_action += 1
107        return new_state
108    
109    def isTerminal(self):
110        """Condition on terminal graph
111
112        Returns:
113            bool: State graph is terminal
114        """
115        terminal_nodes = [node[1]["Node"].is_terminal for node in self.graph.nodes.items()]
116        return sum(terminal_nodes) == len(terminal_nodes)

120    def getReward(self):
121        """Reward in number (3) of nodes graph mechanism
122
123        Returns:
124            float: Reward of terminal state graph
125        """
126        nodes = [node[1]["Node"] for node in self.graph.nodes.items()]
127        self.reward = 10 if len(nodes) == 4 else 0
128        return self.reward

132        """Move current environment to new state
133
134        Args:
135            action (RuleAction): Action is take
136            render (bool): Turn on render each step. Defaults to False.
137
138        Returns:
139            bool, GraphGrammar: Return state of graph. If it is terminal then finish generate graph and new state graph.
140        """
141        new_state = self.takeAction(action)
142        self.graph = new_state.graph
143        self.reward = new_state.reward
144        self.counter_action = new_state.counter_action
145        done = new_state.isTerminal()
146        
147        if render:
148            plt.figure()
149            nx.draw_networkx(self.graph, pos=nx.kamada_kawai_layout(self.graph, dim=2), node_size=800,
150                 labels={n: self.graph.nodes[n]["Node"].label for n in self.graph})
151            plt.show()
152        return done, self.graph

279    def __deepcopy__(self, memo):
280        cls = self.__class__
281        result = cls.__new__(cls)
